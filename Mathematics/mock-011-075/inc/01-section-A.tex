\textbf{Question 1}\\
Choose the correct answers to the questions from the given options. \hfill [7]\\
(Do not copy the questions, write the correct answers only.)
\par
\vspace{2mm}
\begin{enumerate}[label=(\roman*)]

    % ------- Question 1.i ----------------------------------
    % ~   (i) - (c) 
    % ~       [MLQB - Page 76 - Q 12]
    \item The roots of the equation $21x^2 - 8x - 4 = 0$ are:

        \begin{multicols}{2}
        \begin{enumerate}[label=(\alph*)]
            \item $-\dfrac{1}{2}$, $\dfrac{1}{7}$ 
            \item $\dfrac{1}{3}$, $\dfrac{4}{9}$ 
            \item $\dfrac{2}{3}$, $-\dfrac{2}{7}$ 
            \item $\dfrac{4}{7}$, $-\dfrac{2}{3}$ 
        \end{enumerate}
        \end{multicols}

    % ------- Question 1.ii ----------------------------------
    % ~   (ii) - (c) 
    % ~       [MLQB - Page 157 - Q 4]
    \newpage
    \item If the fifth and sixth terms of an A.P. are 6 and 5 
        respectively, then the first term is:

        \begin{multicols}{2}
        \begin{enumerate}[label=(\alph*)]
            \item -1 
            \item 1 
            \item 10 
            \item 20
        \end{enumerate}
        \end{multicols}

    % ------- Question 1.iii ----------------------------------
    % ~   (iii) - (b)
    % ~       [MLQB - Page 354 - Q 12]
    \item If $\sin \theta = \dfrac{1}{3}$, then find the value of 
        $2 \cot^2 \theta + 2$:

        \begin{multicols}{2}
        \begin{enumerate}[label=(\alph*)]
            \item 16 
            \item 18 
            \item 20
            \item 12
        \end{enumerate}
        \end{multicols}

    % ------- Question 1.iv ----------------------------------
    % ~   (iv) - (c)
    % ~       [MLQB - Page 128 - Q 12]
    \item What number should be added to the polynomial 
        $2x^3 - 3x^2 - 8x$ so that the resulting polynomial
        leaves the remainder 12 when divided by $2x+1$?

        \begin{multicols}{2}
        \begin{enumerate}[label=(\alph*)]
            \item 3
            \item 6
            \item 9
            \item 12
        \end{enumerate}
        \end{multicols}

    % ------- Question 1.v ----------------------------------
    % ~   (v) - (b)
    % ~       [MLQB - Page 104 - Q 6]
    \item If $4x = 7y = 9z$, then $x:y:z=$ 

        \begin{multicols}{2}
        \begin{enumerate}[label=(\alph*)]
            \item $24:31:19$ 
            \item $63:36:28$ 
            \item $4:7:9$ 
            \item $16:49:81$ 
        \end{enumerate}
        \end{multicols}

    % ------- Question 1.vi ----------------------------------
    % ~   (vi) - (b)
    % ~       [MLQB - Page 74 - Q 3]
    \item The equation $x^2 + 2x + 1 = (4-kx)^2 + 3$ will be quadratic,
        if the value of $k$ is:

        \begin{multicols}{2}
        \begin{enumerate}[label=(\alph*)]
            \item $k=1$ 
            \item $k \ne \pm 1$ 
            \item Any number
            \item Insufficient data
        \end{enumerate}
        \end{multicols}

    % ------- Question 1.vii ----------------------------------
    % ~   (vii) - (c)
    % ~       [MLQB - Page 41 - Q 9]
    \item Mohan deposits \rupee~80 per month in a cumulative deposit account
        for six years. Find the amount payable to him on maturity, if the rate 
        of interest is 6\% per annum.

        \begin{multicols}{2}
        \begin{enumerate}[label=(\alph*)]
            \item \rupee~6,118.50 
            \item \rupee~6,818.20 
            \item \rupee~6,811.20
            \item \rupee~6,818.50
        \end{enumerate}
        \end{multicols}

\end{enumerate}
