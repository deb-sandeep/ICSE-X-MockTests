    \item If $\sin \theta = \dfrac{12}{13}$, then find the value of $
            \dfrac{2 \sin \theta - 3 \cos \theta}{4 \sin \theta - 9 \cos \theta} $

        \begin{multicols}{2}
        \begin{enumerate}[label=(\alph*)]
            \item 1
            \item 3
            \item 5
            \item 7
        \end{enumerate}
        \end{multicols}

    % ------- Question 1.iii ----------------------------------
    % ~   (iii) - (d) -1/4 
    % ~       [MLQB - Page 74 - Q 4]
    \item If $x=2$ is a solution of the quadratic equation $kx^2 + 2x - 3=0$, then
        the value of $k$ is:

        \begin{multicols}{2}
        \begin{enumerate}[label=(\alph*)]
            \item -1
            \item -4
            \item $\dfrac14$
            \item $-\dfrac14$
        \end{enumerate}
        \end{multicols}

    % ------- Question 1.iv ----------------------------------
    % ~   (iv) - (a) 1:8
    % ~       [MLQB - Page 103 - Q 1]
    \item If $x:y = 2:3$, then the value of $(7x-4y):(5x+2y)$ is: 

        \begin{multicols}{2}
        \begin{enumerate}[label=(\alph*)]
            \item 1:8 
            \item 2:3
            \item 4:9
            \item 5:7
        \end{enumerate}
        \end{multicols}

    % ------- Question 1.v ----------------------------------
        % ~   (v) - (a) [[4, -4], [8, -8]]
    % ~       [MLQB - Page 141 - Q 18]
    \item If $A = \begin{bmatrix*}[r] 4 & -1 \\ 2 & 1 \end{bmatrix*}$, then 
        $(A + 2I)(A-3I)$ is equal to:

        \begin{multicols}{2}
        \begin{enumerate}[label=(\alph*)]
            \item $\begin{bmatrix*}[r] 4 & -4 \\ 8  & -8 \end{bmatrix*}$
            \item $\begin{bmatrix*}[r] 4 &  8 \\ -4 & -8 \end{bmatrix*}$
            \item $\begin{bmatrix*}[r] -4 & 4 \\ -8 &  8 \end{bmatrix*}$
            \item $\begin{bmatrix*}[r] -4 & 4 \\ 8  & -8 \end{bmatrix*}$
        \end{enumerate}
        \end{multicols}

    % ------- Question 1.vi ----------------------------------
    % ~   (vi) - (c) -3 <= x <= 4

            \item $-3 < x \leq 4$
            \item $-3 \leq x \leq 4$
            \item $-3 \leq x < 4$
        \end{enumerate}
        \end{multicols}

    % ------- Question 1.vii ----------------------------------
    % ~   (vii) - (a) Rs. 40,440

            \item \rupee~48,000 
            \item \rupee~44,400 
            \item \rupee~48,040 
        \end{enumerate}
        \end{multicols}

\end{enumerate}
